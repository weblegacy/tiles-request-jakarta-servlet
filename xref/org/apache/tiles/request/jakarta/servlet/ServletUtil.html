16  package org.apache.tiles.request.jakarta.servlet;
17  
18  import java.io.IOException;
19  
20  import org.apache.tiles.request.ApplicationAccess;
21  import org.apache.tiles.request.ApplicationContext;
22  import org.apache.tiles.request.Request;
23  import org.apache.tiles.request.RequestWrapper;
24  
25  import jakarta.servlet.ServletContext;
26  import jakarta.servlet.ServletException;
27  
28  /**
29   * Utilities for Tiles request servlet support.
30   *
31   * <p>Copied from Apache tiles-request-servlet 1.0.7 and adapted for
32   * Jakarta EE 9.</p>
33   */
34  public final class ServletUtil {
35  
36      /**
37       * Constructor.
38       */
39      private ServletUtil() {
40      }
41  
42      /**
43       * Wraps a ServletException to create an IOException with the root cause if
44       * present.
45       *
46       * @param ex      The exception to wrap.
47       * @param message The message of the exception.
48       *
49       * @return The wrapped exception.
50       */
51      public static IOException wrapServletException(ServletException ex,
52              String message) {
53          IOException retValue;
54          Throwable rootCause = ex.getRootCause();
55          if (rootCause != null) {
56              // Replace the ServletException with an IOException, with the root
57              // cause of the first as the cause of the latter.
58              retValue = new IOException(message, rootCause);
59          } else {
60              retValue = new IOException(message, ex);
61          }
62  
63          return retValue;
64      }
65  
66      /**
67       * Returns the application context getting it from the servlet context. It
68       * must be first saved creating a {@link ServletApplicationContext} and
69       * using {@link ApplicationAccess#register(ApplicationContext)}.
70       *
71       * @param servletContext The servlet context.
72       *
73       * @return The application context, if found, {@code null} otherwise.
74       */
75      public static ApplicationContext getApplicationContext(
76              ServletContext servletContext) {
77          return (ApplicationContext) servletContext
78                  .getAttribute(ApplicationAccess.APPLICATION_CONTEXT_ATTRIBUTE);
79      }
80  
81      /**
82       * Opens a TilesRequestContext until it finds a ServletTilesRequestContext.
83       *
84       * @param request The request to open.
85       *
86       * @return The servlet-based request context.
87       *
88       * @throws NotAServletEnvironmentException If a servlet-based request
89       *                                         context could not be found.
90       */
91      public static ServletRequest getServletRequest(Request request) {
92          Request currentRequest = request;
93          while (true) {
94              if (currentRequest == null) {
95                  throw new NotAServletEnvironmentException(
96                          "Last Tiles request context is null");
97              }
98  
99              if (currentRequest instanceof ServletRequest) {
100                 return (ServletRequest) currentRequest;
101             }
102             if (!(currentRequest instanceof RequestWrapper)) {
103                 throw new NotAServletEnvironmentException(
104                         "Not a Servlet environment, not supported");
105             }
106             currentRequest = ((RequestWrapper)
107                     currentRequest).getWrappedRequest();
108         }
109     }
110 
111     /**
112      * Gets a servlet context from a TilesApplicationContext.
113      *
114      * @param applicationContext The application context to analyze.
115      *
116      * @return The servlet context.
117      *
118      * @throws NotAServletEnvironmentException If the application context is not
119      *                                         servlet-based.
120      */
121     public static ServletContext getServletContext(
122             ApplicationContext applicationContext) {
123         if (applicationContext instanceof ServletApplicationContext) {
124             return (ServletContext) ((ServletApplicationContext)
125                     applicationContext).getContext();
126         }
127 
128         throw new NotAServletEnvironmentException(
129                 "Not a Servlet-based environment");
130     }
131 }
